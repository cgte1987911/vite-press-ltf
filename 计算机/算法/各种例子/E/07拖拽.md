```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>
<script>
    window.onload=function(){
        let oWrap=document.querySelector('.wrap')
        let oDiv=document.querySelector('.div')
        dragLimit(oWrap,oDiv)
    }
    function dragLimit(wrap,obj){
        let borderLeft=wrap.offsetLeft
        let borderTop=wrap.offsetTop
        let borderW=wrap.offsetWidth;
        let borderH=wrap.offsetHeight;
        obj.onmousedown=function(ev){
            let offsetX=ev.clientX-this.offsetLeft-borderLeft;
            let offsetY=ev.clientY-this.offsetTop-borderTop;
            document.onmousemove=function(ev){
                let l=ev.clientX-offsetX
                let t=ev.clientY-offsetY
                console.log(l,t)
                if(l<borderLeft){
                    l=borderLeft
                }
                if(t<borderTop){
                    t=borderTop
                }
                if(l>borderLeft+borderW-obj.offsetWidth){
                    l=borderLeft+borderW-obj.offsetWidth
                }
                if(t>borderTop+borderH-obj.offsetHeight){
                    t=borderTop+borderH-obj.offsetHeight
                }
                obj.style.left=l-borderLeft+'px'
                obj.style.top=t-borderTop+'px'
            }
            obj.onmouseup=function(){
                document.onmousemove=null
            }
        }
    }
</script>
<body>
    <div class="wrap" style="width:300px;height:300px;margin:100px auto;position:relative;border:1px solid #ccc;box-sizing: border-box;">
        <div class="div" style="width:100px;height:100px;background:red;position:absolute;left:0;top:0;"></div>
    </div>
</body>
</html>
```
