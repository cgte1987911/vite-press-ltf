```html
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
    <style>
        .bk-line {
            background: -webkit-linear-gradient(45deg, blue 0, red 50%, yellow 100%);
        }

        .out-div {
            overflow:hidden;
            position: relative;
            float: left;
            margin-right: 30px;
            width: 400px;
            height: 400px;
        }

        .mask {
            width: 200px;
            height: 200px;
            background: yellow;
            opacity: 0.5;
            position: absolute;
            left: 0;
            top: 0;
        }

        .cover {
            position: absolute;
            left: 0;
            top: 0;
            z-index: 9;
            width: 100%;
            height: 100%;
        }

        .out-div:nth-of-type(2) div {
            width: 1000px;
            height: 1000px;
            position: absolute;
            left: 0;
            top: 0;
        }
    </style>
</head>

<body>
    <div class="bk-line out-div">
        <div class="mask"></div>
        <div class="cover"></div>
    </div>
    <div class="out-div" style="display:none;">
        <div class="bk-line"></div>
    </div>

    <script>
        window.onload = function () {
            let oDiv = document.querySelector('.out-div')
            let oDiv2 = document.querySelectorAll('.out-div')[1]
            let img = document.querySelectorAll('.out-div:nth-of-type(2) div')[0]
            let oMask = document.querySelector('.mask')

            oDiv.onmouseenter = function () {
                oDiv2.style.display = "block";
                oMask.style.display = 'block';
            }
            oDiv.onmouseleave = function () {
                oDiv2.style.display = "none";
                oMask.style.display = 'none';
            }
            oDiv.onmousemove = function (ev) {
                let left=ev.offsetX-oMask.offsetWidth/2
                let top=ev.offsetY-oMask.offsetHeight/2
                if(left<0){
                    left=0;
                }else if(left>oDiv.offsetWidth-oMask.offsetWidth){
                    left=oDiv.offsetWidth-oMask.offsetWidth;
                }
                if(top<0){
                    top=0;
                }else if(top>oDiv.offsetHeight-oMask.offsetHeight){
                    top=oDiv.offsetHeight-oMask.offsetHeight;
                }
                oMask.style.left=left+'px'
                oMask.style.top=top+'px'

                let x=left*(oDiv2.offsetWidth-img.offsetWidth)/(oDiv.offsetWidth-oMask.offsetWidth)
                let y=top*(oDiv2.offsetHeight-img.offsetHeight)/(oDiv.offsetHeight-oMask.offsetHeight)

                img.style.left=x+'px'
                img.style.top=y+'px'
                
            }
        }
    </script>
</body>

</html>
```
