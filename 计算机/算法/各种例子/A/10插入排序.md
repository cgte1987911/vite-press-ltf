template<typename T>
void insertionSort(T arr[],int n){
    for(int i=1;i<n;i++){
        for(int j=i;j>0;j--){
            if(arr[j]<arr[j-1])
                swap(arr[j],arr[j-1]);
            else
                break;
        }
    }
}

template<typename T>
void insertionSort2(T arr[],int n){
    for(int i=1;i<n;i++){
        T e=arr[i];
        int j=0;
        for( j=i;j>0&&arr[j-1]>e;j--){
            arr[j]=arr[j-1];
        }
        arr[j]=e;
    }
}
