## Student.h
```cpp
#ifndef ltf_Student
#define ltf_Student
#include <iostream>
#include <string>
using namespace std;
struct Student{
    string name;
    int score;

    bool operator <(Student otherStudent){
        return score!=otherStudent.score?score>otherStudent.score:name<otherStudent.name;
    }
    bool operator >(Student otherStudent){
        return score!=otherStudent.score?score<otherStudent.score:name>otherStudent.name;
    }
    friend ostream& operator<<(ostream&os,Student student){
        os<<"name "<<student.name<<" score "<<student.score<<endl;
            return os;
    }
};
#endif
```

#include<iostream>
#include<algorithm>
#include"Student.h"
using namespace std;

template<typename T>
void selectionSort(T arr[],int n){
    int minIndex=0;
    for(int i=0;i<n;i++){
        minIndex=i;
        for(int j=i+1;j<n;j++)
            if(arr[j]<arr[minIndex])
                minIndex=j;
        swap(arr[minIndex],arr[i]);
    }
}

int main(){
    Student D[]={{"D",100},{"C",90},{"B",95},{"A",95}};

    }
}
