        let M=5,W=10
        let ws=[2,2,3,3,4],vs=[3,4,3,4,5]
        function show(M,W,ws,vs){
            let res=[]
            for(let i=0;i<=M;i++){
                res[i]=[]
                for(let j=0;j<=W;j++){
                    if(i==0)
                        res[i][j]=0
                    else if(ws[i-1]>j)
                        res[i][j]=res[i-1][j]
                    else
                        res[i][j]=Math.max(res[i-1][j],vs[i-1]+res[i-1][j-ws[i-1]])
                }
            }
            return res[M][W]
        }
        console.log(show(M,W,ws,vs))
    </script>
