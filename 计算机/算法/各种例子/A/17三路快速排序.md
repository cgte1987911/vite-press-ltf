template <typename T>
void insertionSort(T arr[],int l,int r){
    for(int i=l+1;i<=r;i++){
        T e=arr[i];
        int j;
        for(j=i;j>l&&arr[j-1]>e;j--){
            arr[j]=arr[j-1];
        }
        arr[j]=e;
    }
}
template <typename T>
void __quickSort3Ways(T arr[],int l,int r){
    if(r-l<=15){
        insertionSort(arr,l,r);
        return;
    }
    swap(arr[l],arr[rand()%(r-l+1)+l]);
    T e=arr[l];
    int lt=l;
    int gt=r+1;
    int i=l+1;
    while(i<gt){
        if(arr[i]<e){
            swap(arr[i],arr[lt+1]);
            lt++;
            i++;
        }else if(arr[i]>e){
            swap(arr[i],arr[gt-1]);
            gt--;
        }else{
            i++;
        }
    }
    swap(arr[l],arr[lt]);
    __quickSort3Ways(arr,l,lt-1);
    __quickSort3Ways(arr,gt,r);
}

template<typename T>
void quickSort3Ways(T arr[],int n){
    srand(time(NULL));
    __quickSort3Ways(arr,0,n-1);
}
