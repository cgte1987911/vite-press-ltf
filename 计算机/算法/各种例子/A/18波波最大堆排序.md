template<typename Item>
class MaxHeap{
private:
    Item*data;
    int count;
    int capacity;
public:
    MaxHeap(int capacity){
        data=new Item[capacity+1];
        count=0;
        this->capacity=capacity;
    }
    MaxHeap(Item arr[],int n){
        data=new Item[n+1];
        capacity=n;
        for(int i=0;i<n;i++){
            data[i+1]=arr[i];
        }
        count=n;
        for(int i=n/2;i>=1;i--){
            shiftDown(i);
        }
    }
    ~MaxHeap(){
        delete[] data;
    }
    void insert(Item item){
        assert(count+1<=capacity);
        data[count+1]=item;
        shiftUp(count+1);
        count++;
    }
    Item extractMax(){
        assert(count>=1);
        Item ret=data[1];
        swap(data[1],data[count]);
        count--;
        shiftDown(1);
        return ret;
    }
    void shiftUp(int i){
        while(i/2>0&&data[i/2]<data[i]){
            swap(data[i/2],data[i]);
            i/=2;
        }
    }
    void shiftDown(int i){
        while(i*2<=count){
            int j=i*2;
            if(j+1<=count&&data[j]<data[j+1]){
                j++;
            }
            if(data[j]>data[i]){
                swap(data[i],data[j]);
                i=j;
            }else{
                break;
            }
        }
    }
};
template<typename T>
void heapSort2(T arr[],int n){
    MaxHeap<T>maxheap=MaxHeap<T>(arr,n);
    for(int i=1;i<=n;i++){
        arr[n-i]=maxheap.extractMax();
    }
}
template<typename T>
void shiftDown(T arr[],int n,int k){
    T e=arr[k];
    while(2*k+1<n){
        int j=2*k+1;
        if(2*k+2<n&&arr[j]<arr[j+1])
            j++;
        if(e<arr[j]){
            arr[k]=arr[j];
            k=j;
        }else break;
    }
    arr[k]=e;
}
template <typename T>
void heapSort1(T arr[],int n){
    for(int i=(n-1-1)/2;i>=0;i--){
        shiftDown(arr,n,i);
    }
    for(int i=0;i<n;i++){
        swap(arr[0],arr[n-i-1]);
        shiftDown(arr,n-i-1,0);
    }
}
