#include<iostream>
#include<vector>
using namespace std;
class Solution{
public:
    void moveZeroes(vector<int>&nums){
        int k=0;
        for(int i=0;i<nums.size();i++)
            if(nums[i])
                if(k!=i)
                    swap(nums[i],nums[k++]);
                else
                    k++;
    }
};

int main(){
    int arr[]={0,1,2,0,1,3};
    vector<int>vec(arr,arr+sizeof(arr)/sizeof(int));
    Solution().moveZeroes(vec);
    for(int i=0;i<vec.size();i++)
        cout<<vec[i];
    cout<<endl;
    return 0;
}
