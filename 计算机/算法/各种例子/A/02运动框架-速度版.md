```html
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>
<script>
    window.onload = function () {
        let oDiv = document.querySelector('div')
        startMove(oDiv, {
            left:300,
            top:300,
            opacity:50
        }, () => { alert(1) })
    }

    function startMove(ele,json,fn){
        clearInterval(ele.timer)
        ele.timer=setInterval(()=>{
            let bStop=true;
            for(let attr in json){
                let iCur=0;
                if(attr=='opacity')
                    iCur=Math.round(parseFloat(getComputedStyle(ele,false)[attr]*100));
                else
                    iCur=parseInt(getComputedStyle(ele,false)[attr])
                let speed=(json[attr]-iCur)/6
                speed=speed>0?Math.ceil(speed):Math.floor(speed)
                if(json[attr]!=iCur)
                    bStop=false
                if('opacity'==attr)
                    ele.style[attr]=(speed+iCur)/100
                else
                    ele.style[attr]=iCur+speed+'px'
            }
            if(bStop){
                clearInterval(ele.timer)
                if(fn)
                    fn.call(ele)
            }
        },16)
    }
</script>

<body>
    <div style="width:100px;height:100px;background:red;position:absolute;left :0;top:0;"></div>
</body>

</html>
```
