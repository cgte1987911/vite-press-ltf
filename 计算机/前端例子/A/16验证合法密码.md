```html
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>

<body>
    <script>
        /*
            同时包含数字和小写字母
            • 同时包含数字和大写字母
            • 同时包含小写字母和大写字母
            • 同时包含数字、小写字母和大写字母
            • 以上的 4 种情况是或的关系（实际上，可以不用第 4 条）。
        */

        //解法一：
        var regex = /((?=.*[0-9])(?=.*[a-z])|(?=.*[0-9])(?=.*[A-Z])|(?=.*[a-z])(?=.*[A-Z]))^ [0 - 9A - Za - z]{ 6, 12 } $ /;
        console.log(regex.test("1234567")); // false 全是数字
        console.log(regex.test("abcdef")); // false 全是小写字母
        console.log(regex.test("ABCDEFGH")); // false 全是大写字母
        console.log(regex.test("ab23C")); // false 不足6位
        console.log(regex.test("ABCDEF234")); // true 大写字母和数字
        console.log(regex.test("abcdEF234")); // true 三者都有


        //解法二：
        var regex2 = /(?!^[0-9]{6,12}$)(?!^[a-z]{6,12}$)(?!^[A-Z]{6,12}$)^[0-9A-Za-z]{6,12}$/;
        console.log(regex2.test("1234567")); // false 全是数字
        console.log(regex2.test("abcdef")); // false 全是小写字母
        console.log(regex2.test("ABCDEFGH")); // false 全是大写字母
        console.log(regex2.test("ab23C")); // false 不足6位
        console.log(regex2.test("ABCDEF234")); // true 大写字母和数字
        console.log(regex2.test("abcdEF234")); // true 三者都有
    </script>
</body>

</html>
```
