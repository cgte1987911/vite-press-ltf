        //正向断言
        var result = "hello".replace(/(?=l)/g, '#');
        console.log(result);
        // => "he#l#lo"


        //负向断言

        var result2 = "hello".replace(/(?!l)/g, '#');
        console.log(result2);
        // => "#h#ell#o#"
    </script>
</body>

</html>
