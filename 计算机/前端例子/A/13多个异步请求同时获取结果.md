## 1.js
```js
let fs=require("fs");
let school={}
function after(times,callback){
    return function(){
        if(--times==0){
            callback();
        }
    }
}
const cb=after(2,function(){
    console.log(school)
})
fs.readFile('./name.txt','utf8',function(err,data){
    school.name=data
    cb();
})
fs.readFile('./age.txt','utf8',function(err,data){
    school.age=data
    cb();
})
```
