         */
        var regex = /^([01][0-9]|[2][0-3]):[0-5][0-9]$/;
        console.log(regex.test("23:59"));
        console.log(regex.test("02:07"));

        //二.正则匹配YYYY-MM-DD
        var regex2 = /^[0-9]{4}-(0[1-9]|1[0-2])-(0[1-9]|[12][0-9]|3[01])$/;
        console.log(regex2.test("2017-06-10"));
    </script>
